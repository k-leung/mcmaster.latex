%	KL: Change \chapter command to add an entry into the LoT and LoF for  %
%	each chapter.															%
%																			%
%   Adopted from:  http://texblog.org/tag/addcontentsline/					%
%======																		%
\let\oldchapter\chapter	%													%
\newcommand{\newchapter}[1]{%												%
    \chapter{#1}%															%
    \addcontentsline{lof}{chapter}{%										%
		#1 \vspace{10pt}%													%
    }%																		%
    \addcontentsline{lot}{chapter}{%										%
        #1 \vspace{10pt}%													%
    }%																		%
}%																			%
%=========================================================================%

%=========================================================================%
%	KL: Add a new \newappendix command to add an entries 					%
%		into the LoT and LoF for the appendices							    %
%																			%
%   Adopted from:  http://texblog.org/tag/addcontentsline/					%
%======																		%
\newcommand{\newappendix}[1]{%												%
    \chapter{#1}%															%
	\addcontentsline{lof}{chapter}{%										%
	    Appendix~\thechapter : #1 \vspace{10pt}%							%
    }%																		%
    \addcontentsline{lot}{chapter}{%										%
        Appendix~\thechapter : #1 \vspace{10pt}%							%
    }%																		%
}%																			%
%=========================================================================%
